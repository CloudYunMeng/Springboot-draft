POST /api/auth/register - Register new user (no auth required)
curl -X POST http://localhost:8080/api/auth/register -H "Content-Type: application/json" -d '{"name":"John","email":"john@example.com","password":"pass123","age":25}'
POST /api/auth/login - Login user (no auth required)
curl -X POST http://localhost:8080/api/auth/login -H "Content-Type: application/json" -d '{"email":"john@example.com","password":"pass123"}'
GET /api/users - Get all users (basic auth required)
curl -u "john@example.com:pass123" http://localhost:8080/api/users
